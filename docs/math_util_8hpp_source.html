122constexpr double ipow(const double base, const int expo) {
123 return (expo == 0) ? 1
124 : expo == 1 ? base
125 : expo > 1 ? ((expo & 1) ? base * ipow(base, expo - 1)
126 : ipow(base, expo / 2) * ipow(base, expo / 2))
127 : 1 / ipow(base, -expo);
128}
