173/*In color conversations:
174 * - When converting to bigger color type the LSB weight of 1 LSB is calculated
175 * E.g. 16 bit Red has 5 bits
176 * 8 bit Red has 2 bits
177 * ----------------------
178 * 8 bit red LSB = (2^5 - 1) / (2^2 - 1) = 31 / 3 = 10
179 *
180 * - When calculating to smaller color type simply shift out the LSBs
181 * E.g. 8 bit Red has 2 bits
182 * 16 bit Red has 5 bits
183 * ----------------------
184 * Shift right with 5 - 3 = 2
185 */
186
187static inline uint8_t lv_color_to1(lv_color_t color)
188{
189#if LV_COLOR_DEPTH == 1
190 return color.full;
191#elif LV_COLOR_DEPTH == 8
192 if((color.red & 0x4) ||
193 (color.green & 0x4) ||
194 (color.blue & 0x2)) {
195 return 1;
196 } else {
197 return 0;
198 }
199#elif LV_COLOR_DEPTH == 16
200# if LV_COLOR_16_SWAP == 0
201 if((color.red & 0x10) ||
202 (color.green & 0x20) ||
203 (color.blue & 0x10)) {
204 return 1;
205# else
206 if((color.red & 0x10) ||
207 (color.green_h & 0x20) ||
208 (color.blue & 0x10)) {
209 return 1;
210# endif
211 } else {
212 return 0;
213 }
214#elif LV_COLOR_DEPTH == 32
215 if((color.red & 0x80) ||
216 (color.green & 0x80) ||
217 (color.blue & 0x80)) {
218 return 1;
219 } else {
220 return 0;
221 }
222#endif
223}

225static inline uint8_t lv_color_to8(lv_color_t color)
226{
227#if LV_COLOR_DEPTH == 1
228 if(color.full == 0) return 0;
229 else return 0xFF;
230#elif LV_COLOR_DEPTH == 8
231 return color.full;
232#elif LV_COLOR_DEPTH == 16
233
234# if LV_COLOR_16_SWAP == 0
235 lv_color8_t ret;
236 ret.red = color.red >> 2; /* 5 - 3 = 2*/
237 ret.green = color.green >> 3; /* 6 - 3 = 3*/
238 ret.blue = color.blue >> 3; /* 5 - 2 = 3*/
239 return ret.full;
240# else
241 lv_color8_t ret;
242 ret.red = color.red >> 2; /* 5 - 3 = 2*/
243 ret.green = color.green_h; /* 6 - 3 = 3*/
244 ret.blue = color.blue >> 3; /* 5 - 2 = 3*/
245 return ret.full;
246# endif
247#elif LV_COLOR_DEPTH == 32
248 lv_color8_t ret;
249 ret.red = color.red >> 5; /* 8 - 3 = 5*/
250 ret.green = color.green >> 5; /* 8 - 3 = 5*/
251 ret.blue = color.blue >> 6; /* 8 - 2 = 6*/
252 return ret.full;
253#endif
254}

256static inline uint16_t lv_color_to16(lv_color_t color)
257{
258#if LV_COLOR_DEPTH == 1
259 if(color.full == 0) return 0;
260 else return 0xFFFF;
261#elif LV_COLOR_DEPTH == 8
262 lv_color16_t ret;
263# if LV_COLOR_16_SWAP == 0
264 ret.red = color.red * 4; /*(2^5 - 1)/(2^3 - 1) = 31/7 = 4*/
265 ret.green = color.green * 9; /*(2^6 - 1)/(2^3 - 1) = 63/7 = 9*/
266 ret.blue = color.blue * 10; /*(2^5 - 1)/(2^2 - 1) = 31/3 = 10*/
267# else
268 ret.red = color.red * 4;
269 uint8_t g_tmp = color.green * 9;
270 ret.green_h = (g_tmp & 0x1F) >> 3;
271 ret.green_l = g_tmp & 0x07;
272 ret.blue = color.blue * 10;
273# endif
274 return ret.full;
275#elif LV_COLOR_DEPTH == 16
276 return color.full;
277#elif LV_COLOR_DEPTH == 32
278 lv_color16_t ret;
279# if LV_COLOR_16_SWAP == 0
280 ret.red = color.red >> 3; /* 8 - 5 = 3*/
281 ret.green = color.green >> 2; /* 8 - 6 = 2*/
282 ret.blue = color.blue >> 3; /* 8 - 5 = 3*/
283# else
284 ret.red = color.red >> 3;
285 ret.green_h = (color.green & 0xE0) >> 5;
286 ret.green_l = (color.green & 0x1C) >> 2;
287 ret.blue = color.blue >> 3;
288# endif
289 return ret.full;
290#endif
291}

293static inline uint32_t lv_color_to32(lv_color_t color)
294{
295#if LV_COLOR_DEPTH == 1
296 if(color.full == 0) return 0;
297 else return 0xFFFFFFFF;
298#elif LV_COLOR_DEPTH == 8
299 lv_color32_t ret;
300 ret.red = color.red * 36; /*(2^8 - 1)/(2^3 - 1) = 255/7 = 36*/
301 ret.green = color.green * 36; /*(2^8 - 1)/(2^3 - 1) = 255/7 = 36*/
302 ret.blue = color.blue * 85; /*(2^8 - 1)/(2^2 - 1) = 255/3 = 85*/
303 ret.alpha = 0xFF;
304 return ret.full;
305#elif LV_COLOR_DEPTH == 16
306# if LV_COLOR_16_SWAP == 0
307 lv_color32_t ret;
308 ret.red = color.red * 8; /*(2^8 - 1)/(2^5 - 1) = 255/31 = 8*/
309 ret.green = color.green * 4; /*(2^8 - 1)/(2^6 - 1) = 255/63 = 4*/
310 ret.blue = color.blue * 8; /*(2^8 - 1)/(2^5 - 1) = 255/31 = 8*/
311 ret.alpha = 0xFF;
312 return ret.full;
313# else
314 lv_color32_t ret;
315 ret.red = color.red * 8; /*(2^8 - 1)/(2^5 - 1) = 255/31 = 8*/
316 ret.green = ((color.green_h << 3) + color.green_l) * 4; /*(2^8 - 1)/(2^6 - 1) = 255/63 = 4*/
317 ret.blue = color.blue * 8; /*(2^8 - 1)/(2^5 - 1) = 255/31 = 8*/
318 ret.alpha = 0xFF;
319 return ret.full;
320# endif
321#elif LV_COLOR_DEPTH == 32
322 return color.full;
323#endif
324}

326static inline lv_color_t lv_color_mix(lv_color_t c1, lv_color_t c2, uint8_t mix)
327{
328 lv_color_t ret;
329#if LV_COLOR_DEPTH != 1
330 /*LV_COLOR_DEPTH == 8, 16 or 32*/
331 ret.red = (uint16_t)((uint16_t) c1.red * mix + (c2.red * (255 - mix))) >> 8;
332# if LV_COLOR_DEPTH == 16 && LV_COLOR_16_SWAP
333 /*If swapped Green is in 2 parts*/
334 uint16_t g_1 = (c1.green_h << 3) + c1.green_l;
335 uint16_t g_2 = (c2.green_h << 3) + c2.green_l;
336 uint16_t g_out = (uint16_t)((uint16_t) g_1 * mix + (g_2 * (255 - mix))) >> 8;
337 ret.green_h = g_out >> 3;
338 ret.green_l = g_out & 0x7;
339# else
340 ret.green = (uint16_t)((uint16_t) c1.green * mix + (c2.green * (255 - mix))) >> 8;
341# endif
342 ret.blue = (uint16_t)((uint16_t) c1.blue * mix + (c2.blue * (255 - mix))) >> 8;
343# if LV_COLOR_DEPTH == 32
344 ret.alpha = 0xFF;
345# endif
346#else
347 /*LV_COLOR_DEPTH == 1*/
348 ret.full = mix > LV_OPA_50 ? c1.full : c2.full;
349#endif
350
351 return ret;
352}

359static inline uint8_t lv_color_brightness(lv_color_t color)
360{
361 lv_color32_t c32;
362 c32.full = lv_color_to32(color);
363 uint16_t bright = 3 * c32.red + c32.blue + 4 * c32.green;
364 return (uint16_t) bright >> 3;
365}

381#ifdef __cplusplus

390#elif LV_COLOR_DEPTH == 8
