130/*
131Given an id and a port, this macro sets the port
132variable based on the id and allows the mutex to take that port.
133
134Returns error (in the function/scope it's in) if the controller
135failed to connect or an invalid id is given.
136*/
137#define CONTROLLER_PORT_MUTEX_TAKE(id, port) \
138 switch (id) { \
139 case E_CONTROLLER_MASTER: \
140 port = V5_PORT_CONTROLLER_1; \
141 break; \
142 case E_CONTROLLER_PARTNER: \
143 port = V5_PORT_CONTROLLER_2; \
144 break; \
145 default: \
146 errno = EINVAL; \
147 return PROS_ERR; \
148 } \
149 if (!internal_port_mutex_take(port)) { \
150 errno = EACCES; \
151 return PROS_ERR; \
152 } \
